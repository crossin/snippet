def club_comment(df,result):
    clubs = pd.DataFrame(np.zeros(7 * len(df)).reshape(len(df),7),
                      columns = ['C','','','','','',''])
    for i in range(len(result)):
        words = result[i]
        for word in words:
            if '' in word or '' in word:
                clubs.iloc[i + 1][''] = 1
            if 'C' in word or 'C' in word or 'c' in word or '' in word:
                clubs.iloc[i + 1]['C'] = 1
            if '' in word or '' in word:
                clubs.iloc[i + 1][''] = 1
            if '' in word or '' in word or '' in word:
                clubs.iloc[i + 1][''] = 1
            if '' in word or '' in word:
                clubs.iloc[i + 1][''] = 1
            if '' in word or '' in word:
                clubs.iloc[i + 1][''] = 1
            if '' in word or '' in word:
                clubs.iloc[i + 1][''] = 1
            if '' in word:
                clubs.iloc[i + 1][''] = 1
    final_result = pd.concat([df,clubs],axis = 1)
    return final_result
